# RNN 实战：情感分析

## 1️⃣ RNN 的核心：隐藏状态循环

```python
import torch
import torch.nn as nn
import numpy as np

# 手动展开一个 RNN cell
rnn_cell = nn.RNNCell(input_size=10, hidden_size=20)

x = torch.randn(3, 10)   # 3 个时间步，每步 10 维输入
h = torch.zeros(20)       # 初始隐藏状态

print("手动展开 RNN（理解循环本质）:")
for t in range(3):
    h = rnn_cell(x[t], h)  # x[t] + 上一步的 h → 新的 h
    print(f"  t={t}: h sum = {h.sum().item():.4f}")

# PyTorch 帮你自动展开
rnn = nn.RNN(input_size=10, hidden_size=20, num_layers=1, batch_first=True)
x_batch = torch.randn(8, 5, 10)  # batch=8, seq_len=5, input_dim=10
output, hn = rnn(x_batch)
print(f"\nRNN 输出 shape: {output.shape}")  # [8, 5, 20]
print(f"最后一层隐藏状态 shape: {hn.shape}")  # [1, 8, 20]
```

## 2️⃣ LSTM：比 RNN 多了"记忆高速公路"

```python
# LSTM 有三个门 + 一个细胞状态
lstm = nn.LSTM(input_size=10, hidden_size=20, num_layers=2, batch_first=True, bidirectional=True)

x = torch.randn(16, 100, 10)  # batch=16, seq_len=100, dim=10
output, (hn, cn) = lstm(x)

print(f"双向 LSTM 输出: {output.shape}")  # [16, 100, 40] — 因为是双向，hidden*2

print("\nLSTM 三门结构:")
print("  遗忘门 f_t: σ(W_f·[h_{t-1}, x_t])  ← 决定扔掉哪些旧记忆")
print("  输入门 i_t: σ(W_i·[h_{t-1}, x_t])  ← 决定写入哪些新信息")
print("  输出门 o_t: σ(W_o·[h_{t-1}, x_t])  ← 决定暴露哪些记忆")
print("  细胞状态: c_t = f_t⊙c_{t-1} + i_t⊙c̃_t  ← 记忆高速公路")
```

## 3️⃣ 🔥 完整的情感分析模型

```python
class SentimentLSTM(nn.Module):
    def __init__(self, vocab_size, embed_dim=100, hidden_dim=128, num_layers=2, num_classes=2):
        super().__init__()
        self.embedding = nn.Embedding(vocab_size, embed_dim, padding_idx=0)
        self.lstm = nn.LSTM(embed_dim, hidden_dim, num_layers, 
                            batch_first=True, bidirectional=True, dropout=0.3)
        self.fc = nn.Sequential(
            nn.Linear(hidden_dim * 2, 64),  # *2 因为是双向
            nn.ReLU(),
            nn.Dropout(0.5),
            nn.Linear(64, num_classes),
        )
    
    def forward(self, x):
        # x: [batch, seq_len]
        embedded = self.embedding(x)               # [batch, seq_len, embed_dim]
        lstm_out, (hn, cn) = self.lstm(embedded)
        # 取两个方向的最后一层隐藏状态拼接
        last_hidden = torch.cat((hn[-2], hn[-1]), dim=1)  # [batch, hidden*2]
        return self.fc(last_hidden)
```

## 6️⃣ GRU vs LSTM：简化但有效

```python
# GRU 比 LSTM 少一个门，参数少 25%，但效果常常相当
gru = nn.GRU(input_size=10, hidden_size=20, num_layers=2, batch_first=True)

print("GRU 的两个门:")
print("  重置门 r_t: 决定是否忽略旧状态")
print("  更新门 z_t: 决定保留多少旧状态 vs 写入新状态")
print("\n💡 实践中：数据少时用 GRU（参数少），数据多用 LSTM（表达能力强）")
```
