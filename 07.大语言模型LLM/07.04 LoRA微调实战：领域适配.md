## 1️⃣ LoRA 原理：不改变原有权重，只加"补丁"

```
原始: h = W₀ x            (W₀ 冻结，不更新)

LoRA: h = W₀ x + B A x    (只训练 A 和 B)
              ↑   ↑
          d×r  r×d     (r << d，比如 d=4096, r=16)

参数量: 2 × d × r  vs  d × d
压缩比: 2r / d  (r=16, d=4096 → 0.78%)
```

```python
import torch
import torch.nn as nn

class LoRALinear(nn.Module):
    def __init__(self, in_features, out_features, r=16, alpha=32):
        super().__init__()
        self.linear = nn.Linear(in_features, out_features, bias=False)
        # 冻结原始权重
        for p in self.linear.parameters():
            p.requires_grad = False
        
        # LoRA 参数
        self.lora_A = nn.Parameter(torch.randn(r, in_features) * 0.02)
        self.lora_B = nn.Parameter(torch.zeros(out_features, r))
        self.scaling = alpha / r
    
    def forward(self, x):
        W0_out = self.linear(x)
        lora_out = (x @ self.lora_A.T @ self.lora_B.T) * self.scaling
        return W0_out + lora_out

# 参数对比
d = 4096
full_params = d * d
lora_params = 2 * d * 16  # r=16
print(f"全量参数: {full_params:,}")
print(f"LoRA 参数: {lora_params:,}")
print(f"压缩比: {lora_params / full_params:.2%}")
```

## 2️⃣ 🔥 用 PEFT 微调 LLaMA/Qwen（实战代码）

```python
# pip install transformers peft accelerate bitsandbytes datasets

from transformers import (
    AutoModelForCausalLM, AutoTokenizer,
    BitsAndBytesConfig, TrainingArguments, Trainer
)
from peft import LoraConfig, get_peft_model, TaskType
from datasets import Dataset
import torch

# === 1. 量化加载（4-bit 节省显存） ===
bnb_config = BitsAndBytesConfig(
    load_in_4bit=True,
    bnb_4bit_compute_dtype=torch.float16,
    bnb_4bit_quant_type="nf4",
)

model_name = "Qwen/Qwen2-0.5B"  # 小模型，适合学习
tokenizer = AutoTokenizer.from_pretrained(model_name, trust_remote_code=True)
model = AutoModelForCausalLM.from_pretrained(
    model_name,
    quantization_config=bnb_config,
    device_map="auto",
    trust_remote_code=True,
)

# === 2. 配置 LoRA ===
lora_config = LoraConfig(
    task_type=TaskType.CAUSAL_LM,
    r=16,               # 低秩维度
    lora_alpha=32,      # 缩放因子
    lora_dropout=0.1,
    target_modules=["q_proj", "v_proj", "k_proj", "o_proj"],  # Qwen 的 attention 层
)

model = get_peft_model(model, lora_config)
model.print_trainable_parameters()
# 输出: trainable params: ~3M || all params: ~500M || trainable%: 0.6%
```

## 3️⃣ 准备微调数据

```python
# Alpaca 格式（指令微调的标准格式）
training_data = [
    {
        "instruction": "将以下句子翻译成英文",
        "input": "今天天气真好",
        "output": "The weather is really nice today."
    },
    {
        "instruction": "解释以下概念",
        "input": "什么是机器学习？",
        "output": "机器学习是人工智能的一个分支..."
    },
]

def format_alpaca(example):
    """格式化成模型能理解的文本"""
    if example["input"]:
        text = f"""Below is an instruction that describes a task, paired with an input.
        
### Instruction:
{example["instruction"]}

### Input:
{example["input"]}

### Response:
{example["output"]}"""
    else:
        text = f"""### Instruction:
{example["instruction"]}

### Response:
{example["output"]}"""
    return {"text": text}

dataset = Dataset.from_list(training_data).map(format_alpaca)

def tokenize_function(examples):
    result = tokenizer(
        examples["text"],
        truncation=True,
        max_length=512,
        padding="max_length",
    )
    result["labels"] = result["input_ids"].copy()
    return result

tokenized_dataset = dataset.map(tokenize_function, batched=True)
```

## 4️⃣ 训练

```python
training_args = TrainingArguments(
    output_dir="./lora_finetuned",
    num_train_epochs=3,
    per_device_train_batch_size=4,
    gradient_accumulation_steps=4,  # 模拟更大的 batch
    learning_rate=2e-4,
    warmup_ratio=0.03,
    logging_steps=10,
    save_strategy="epoch",
    fp16=True,
    report_to="none",
)

trainer = Trainer(
    model=model,
    args=training_args,
    train_dataset=tokenized_dataset,
)

trainer.train()

# 保存 LoRA 权重（只有几 MB！）
model.save_pretrained("./lora_adapter")
print("LoRA adapter 已保存（通常只有几 MB - 几十 MB）")
```

## 5️⃣ 加载和使用微调后的模型

```python
from peft import PeftModel

# 加载基础模型 + LoRA adapter
base_model = AutoModelForCausalLM.from_pretrained(model_name)
model = PeftModel.from_pretrained(base_model, "./lora_adapter")

# 推理
inputs = tokenizer("### Instruction:\n写一首关于春天的诗\n\n### Response:\n", return_tensors="pt")
outputs = model.generate(**inputs, max_new_tokens=100, temperature=0.7)
print(tokenizer.decode(outputs[0], skip_special_tokens=True))
```
