## 1️⃣ 目标检测的核心概念

```python
import numpy as np
import matplotlib.pyplot as plt
import matplotlib.patches as patches

# IoU = 交集面积 / 并集面积
def compute_iou(box1, box2):
    """box format: [x1, y1, x2, y2]"""
    x1 = max(box1[0], box2[0])
    y1 = max(box1[1], box2[1])
    x2 = min(box1[2], box2[2])
    y2 = min(box1[3], box2[3])
    
    intersection = max(0, x2 - x1) * max(0, y2 - y1)
    area1 = (box1[2] - box1[0]) * (box1[3] - box1[1])
    area2 = (box2[2] - box2[0]) * (box2[3] - box2[1])
    union = area1 + area2 - intersection
    
    return intersection / union if union > 0 else 0

# 可视化 IoU
box_pred = [30, 40, 130, 140]    # 预测框
box_gt = [50, 50, 150, 150]      # 真实框
iou = compute_iou(box_pred, box_gt)

fig, ax = plt.subplots(figsize=(6, 6))
for box, color, label in [(box_gt, 'green', 'Ground Truth'), (box_pred, 'red', 'Prediction')]:
    rect = patches.Rectangle((box[0], box[1]), box[2]-box[0], box[3]-box[1],
                               linewidth=2, edgecolor=color, facecolor='none')
    ax.add_patch(rect)
    ax.annotate(label, (box[0], box[1]-5), color=color)

ax.set_xlim(0, 200); ax.set_ylim(0, 200)
ax.set_title(f'IoU = {iou:.2f}')
ax.set_aspect('equal'); plt.show()

print(f"IoU = {iou:.2f}")
print("IoU > 0.5 → 一般认为是正确检测")
print("mAP = 在不同 IoU 阈值下平均精确率的均值")
```

## 2️⃣ YOLO 的核心思想

```
传统方法（两阶段）：         YOLO（单阶段）：
1. 先找出可能有物体的区域      1. 把图分成 S×S 网格
2. 再在每个区域做分类          2. 每个网格预测 B 个框 + 类别
                              3. 一步到位！
```

```python
# YOLO 的输出格式
# 每个网格预测：
# [x, y, w, h, confidence, class1_prob, class2_prob, ...]
#  x,y = 框中心（相对于网格）
#  w,h = 框宽高（相对于整图）
#  confidence = P(object) × IoU(truth, pred)
```

## 3️⃣ 🔥 Ultralytics YOLO：一行代码跑检测

```python
# pip install ultralytics
from ultralytics import YOLO

# 加载预训练模型
model = YOLO('yolo11n.pt')  # nano 版，最快最小

# 推理 — 一行代码
results = model('https://ultralytics.com/images/bus.jpg', show=True)

# 解析结果
for result in results:
    boxes = result.boxes
    if boxes is not None:
        print(f"检测到 {len(boxes)} 个物体:")
        for box in boxes:
            cls = int(box.cls[0])
            conf = float(box.conf[0])
            xyxy = box.xyxy[0].tolist()
            print(f"  类别 {model.names[cls]:10s} 置信度 {conf:.2f} 位置 {xyxy}")
```

## 4️⃣ 在自己的数据上训练 YOLO

```python
# 1. 准备数据（YOLO 格式）
# 数据集结构：
# dataset/
#   images/
#     train/img1.jpg
#     val/img2.jpg
#   labels/
#     train/img1.txt  ← 每行: class_id x_center y_center width height
#     val/img2.txt

# 2. 创建 data.yaml
yaml_content = """
path: ./dataset
train: images/train
val: images/val
names:
  0: cat
  1: dog
"""

# 3. 训练
model = YOLO('yolo11n.pt')
results = model.train(
    data='data.yaml',
    epochs=50,
    imgsz=640,
    batch=16,
    name='my_detector'
)

# 4. 评估
metrics = model.val()
print(f"mAP@0.5: {metrics.box.map50:.3f}")
print(f"mAP@0.5:0.95: {metrics.box.map:.3f}")
```

## 5️⃣ Anchor Box：YOLO 怎么处理不同形状的物体？

```python
# Anchor = 预定义的"模板框"
# 不同尺度和长宽比的 anchor 负责检测不同类型的物体

# YOLO 默认 anchor（示例）：
anchors = [
    # 小尺度：适合检测小物体
    [10, 13], [16, 30], [33, 23],
    # 中尺度
    [30, 61], [62, 45], [59, 119],
    # 大尺度：适合检测大物体
    [116, 90], [156, 198], [373, 326],
]

print("💡 Anchor 的设计哲学：")
print("   瘦高的 anchor → 检测站立的人")
print("   扁宽的 anchor → 检测汽车")
print("   小的 anchor → 检测远处的物体")
print("   YOLOv5+ 用 AutoAnchor 自动聚类最优尺寸")
```
