## 📝 练习清单

### 1. Matplotlib 基础
```python
import matplotlib.pyplot as plt
import numpy as np

# 折线图
x = np.linspace(0, 10, 100)
plt.plot(x, np.sin(x), label='sin(x)')
plt.plot(x, np.cos(x), label='cos(x)')
plt.xlabel('x')
plt.ylabel('y')
plt.title('Sin and Cos')
plt.legend()
plt.grid(True)
plt.show()
```

### 2. 常用图表类型
```python
# 散点图
plt.scatter(x, y, c='red', alpha=0.5, s=10)

# 柱状图
categories = ['A', 'B', 'C', 'D']
values = [23, 45, 56, 78]
plt.bar(categories, values, color='skyblue')

# 直方图
data = np.random.randn(1000)
plt.hist(data, bins=30, edgecolor='black', alpha=0.7)

# 箱线图
plt.boxplot([np.random.randn(100), np.random.randn(100)])
```

### 3. 子图与布局
```python
fig, axes = plt.subplots(2, 2, figsize=(10, 8))
axes[0, 0].plot(x, np.sin(x))
axes[0, 0].set_title('Sin')
axes[0, 1].plot(x, np.cos(x))
axes[0, 1].set_title('Cos')
axes[1, 0].scatter(x[:20], np.random.randn(20))
axes[1, 1].hist(np.random.randn(1000), bins=30)
plt.tight_layout()
plt.show()
```

### 4. Seaborn 快速统计图
```python
import seaborn as sns

# 加载内置数据集
tips = sns.load_dataset('tips')

# 箱线图
sns.boxplot(x='day', y='total_bill', data=tips)

# 热力图（相关性）
corr = tips[['total_bill', 'tip', 'size']].corr()
sns.heatmap(corr, annot=True, cmap='coolwarm')

# Pairplot（多变量关系矩阵）
sns.pairplot(tips, hue='sex')

# countplot（类别计数）
sns.countplot(x='day', data=tips)
```

## 🔥 综合实战：Titanic 数据可视化

加载清洗后的 Titanic 数据，完成以下图表：

```python
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

df = pd.read_csv('titanic_clean.csv')

# 1. 年龄分布直方图
plt.figure(figsize=(8, 5))
plt.hist(df['Age'], bins=30, edgecolor='black', alpha=0.7)
plt.xlabel('Age')
plt.ylabel('Count')
plt.title('Age Distribution of Titanic Passengers')
plt.show()

# 2. 生存与性别的柱状图
sns.countplot(x='Sex', hue='Survived', data=df)
plt.title('Survival by Gender')
plt.show()

# 3. 票价与舱位等级
sns.boxplot(x='Pclass', y='Fare', data=df)
plt.title('Fare Distribution by Passenger Class')
plt.show()

# 4. 相关性热力图
numeric_cols = df.select_dtypes(include=[np.number]).columns
plt.figure(figsize=(10, 8))
sns.heatmap(df[numeric_cols].corr(), annot=True, cmap='RdBu', center=0)
plt.title('Feature Correlation Heatmap')
plt.show()
```
