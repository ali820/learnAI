## 1️⃣ 概率分布 — 动手感受

```python
import numpy as np
import matplotlib.pyplot as plt
from scipy import stats

# 正态分布 — AI 中最常见的分布
mu, sigma = 0, 1
x = np.linspace(-4, 4, 200)
pdf = stats.norm.pdf(x, mu, sigma)

fig, axes = plt.subplots(1, 2, figsize=(14, 5))

axes[0].plot(x, pdf, 'b-', linewidth=2)
axes[0].fill_between(x, pdf, alpha=0.2)
axes[0].axvline(mu, color='r', linestyle='--', label=f'μ={mu}')
axes[0].axvline(mu-sigma, color='g', linestyle=':', label=f'±1σ')
axes[0].axvline(mu+sigma, color='g', linestyle=':')
axes[0].set_title('标准正态分布 N(0,1)')
axes[0].legend()
axes[0].grid(True, alpha=0.3)

# 从分布中采样
samples = np.random.randn(10000)
axes[1].hist(samples, bins=50, density=True, alpha=0.6, edgecolor='black')
axes[1].plot(x, pdf, 'r-', linewidth=2, label='理论分布')
axes[1].set_title('采样 10000 个点')
axes[1].legend()
axes[1].grid(True, alpha=0.3)

plt.tight_layout()
plt.show()

# 68-95-99.7 法则验证
for k in [1, 2, 3]:
    within = np.mean(np.abs(samples) < k * sigma) * 100
    print(f"在 ±{k}σ 内的数据占比: {within:.1f}%  (理论: {[68.3, 95.4, 99.7][k-1]}%)")
```

> 💡 **为什么 AI 这么爱用正态分布？** 中心极限定理：不管原始数据什么分布，大量独立随机变量的和/均值近似正态。权重初始化、数据标准化、VAE 潜变量都用它。

## 2️⃣ 条件概率与贝叶斯定理 — 手算 + 代码

### 经典问题：医学检测

```
某疾病发病率 P(D) = 0.01（1%）
检测准确率 P(+|D) = 0.95（真有病时测出阳性 95%）
假阳性率 P(+|¬D) = 0.05（没病时测出阳性 5%）

问：检测阳性时，真有病的概率 P(D|+) 是多少？
```

```python
# 贝叶斯公式：P(D|+) = P(+|D)P(D) / P(+)
#                     = P(+|D)P(D) / (P(+|D)P(D) + P(+|¬D)P(¬D))

P_D = 0.01        # 先验（发病率）
P_notD = 1 - P_D
P_pos_given_D = 0.95   # 似然（真有病测出阳性）
P_pos_given_notD = 0.05  # 假阳性率

# 全概率公式
P_pos = P_pos_given_D * P_D + P_pos_given_notD * P_notD

# 贝叶斯公式
P_D_given_pos = P_pos_given_D * P_D / P_pos

print("=== 贝叶斯定理 ===")
print(f"先验 P(D) = {P_D:.3f}")
print(f"似然 P(+|D) = {P_pos_given_D:.3f}")
print(f"证据 P(+) = {P_pos:.4f}")
print(f"\n🔑 后验 P(D|+) = {P_D_given_pos:.4f} = {P_D_given_pos*100:.1f}%")
print(f"\n💡 检测阳性后，真有病的概率只有 {P_D_given_pos*100:.1f}%！")
print(f"   为什么？因为疾病本身太罕见了（先验只有 1%），")
print(f"   即使检测很准，假阳性的人数仍远超真阳性。")
```

> 💡 **这正是贝叶斯推断的核心**：用新证据（检测结果）更新先验信念（发病率），得到后验信念。

## 3️⃣ 🔥 核心实战：朴素贝叶斯垃圾邮件分类器

```python
from collections import defaultdict
import numpy as np

# 构建小型训练集
spam_emails = [
    "免费 赢取 大奖 点击 链接",
    "免费 免费 促销 优惠 立即 购买",
    "恭喜 你 中奖 了 点击 领取",
    "限时 优惠 免费 送货 上门",
    "赚钱 机会 在家 工作 每天",
]

ham_emails = [
    "明天 开会 下午 三点 记得 参加",
    "项目 进展 报告 已经 发到 邮箱",
    "周末 一起 吃饭 怎么样",
    "请 查收 附件 中的 合同",
    "会议 纪要 已经 整理 完毕",
]

# === 训练：计算先验和条件概率 ===

# 1. 先验概率
n_spam = len(spam_emails)
n_ham = len(ham_emails)
n_total = n_spam + n_ham
p_spam = n_spam / n_total
p_ham = n_ham / n_total
print(f"P(spam) = {p_spam:.2f}, P(ham) = {p_ham:.2f}")

# 2. 统计每个词在各类别中的出现次数（加 1 平滑）
word_count_spam = defaultdict(int)
word_count_ham = defaultdict(int)

for email in spam_emails:
    for word in email.split():
        word_count_spam[word] += 1

for email in ham_emails:
    for word in email.split():
        word_count_ham[word] += 1

# 词汇表
vocab = set(word_count_spam.keys()) | set(word_count_ham.keys())
print(f"词汇表大小: {len(vocab)} 个词")

# 3. 条件概率 P(word|class) — 加 1 平滑（拉普拉斯平滑）
total_spam_words = sum(word_count_spam.values())
total_ham_words = sum(word_count_ham.values())

def p_word_given_class(word, class_counts, total_words):
    """P(word|class) = (count + 1) / (total + |V|)"""
    return (class_counts.get(word, 0) + 1) / (total_words + len(vocab))

# === 预测 ===
def predict(email):
    """对一封中文邮件预测是否为垃圾邮件"""
    words = email.split()
    
    # log P(spam) + Σ log P(word|spam)
    log_prob_spam = np.log(p_spam)
    log_prob_ham = np.log(p_ham)
    
    for word in words:
        log_prob_spam += np.log(p_word_given_class(word, word_count_spam, total_spam_words))
        log_prob_ham += np.log(p_word_given_class(word, word_count_ham, total_ham_words))
    
    return "spam" if log_prob_spam > log_prob_ham else "ham"

# === 测试 ===
test_emails = [
    "免费 优惠 点击 链接 领取 大奖",          # 应该是 spam
    "明天 会议 报告 请 查收",                # 应该是 ham
    "项目 进展 免费 免费",                   # ham 内容但有 spam 词
    "在家 赚钱 免费 免费 免费",              # 应该是 spam
]

print("\n=== 预测结果 ===")
for email in test_emails:
    result = predict(email)
    print(f"[{result.upper():4s}] {email}")

# === 看内部概率 ===
print("\n=== 关键条件概率 ===")
for word in ['免费', '会议', '优惠', '报告']:
    p_s = p_word_given_class(word, word_count_spam, total_spam_words)
    p_h = p_word_given_class(word, word_count_ham, total_ham_words)
    print(f"P('{word}'|spam) = {p_s:.4f}, P('{word}'|ham) = {p_h:.4f}")
```

> 💡 **为什么取 log？** 连乘很多 <1 的概率会下溢（趋近 0）。取 log 后变成加法，数值稳定。
> 而且 log 是单调函数，`log(a) > log(b)` 当且仅当 `a > b`，不影响比较结果。

## 4️⃣ 极大似然估计（MLE）的直觉

```python
# 场景：一枚硬币抛 10 次，7 次正面，3 次反面
# 问：硬币正面朝上的概率 θ 最可能是多少？

# 似然函数：P(7正3反 | θ) = θ^7 * (1-θ)^3
# MLE：选使似然最大的 θ

thetas = np.linspace(0.01, 0.99, 100)
likelihood = thetas**7 * (1 - thetas)**3

fig, ax = plt.subplots(figsize=(8, 5))
ax.plot(thetas, likelihood, 'b-', linewidth=2)
theta_opt = 7 / 10  # 理论最优：7/(7+3) = 0.7
ax.axvline(theta_opt, color='r', linestyle='--', label=f'MLE: θ={theta_opt:.1f}')
ax.set_xlabel('θ (硬币正面概率)')
ax.set_ylabel('似然 P(7正3反 | θ)')
ax.set_title('极大似然估计：找让观测数据最可能出现的参数')
ax.legend()
ax.grid(True, alpha=0.3)
plt.show()

print(f"MLE 估计: θ = {theta_opt}")
print(f"直觉：抛了 10 次看到 7 次正面，最合理的估计当然是 7/10 = 0.7")
```

> 💡 **MLE 是大多数损失函数的理论来源。** 最小化 MSE = 假设高斯噪声下的 MLE；最小化交叉熵 = 假设伯努利/类别分布下的 MLE。

## 5️⃣ 交叉熵 = 分类问题的 MLE 损失

```python
# 交叉熵：衡量预测分布与真实分布的差距
# H(p, q) = -Σ p(x) log q(x)

def cross_entropy(y_true, y_pred_proba):
    """
    y_true: 真实标签 (one-hot 或类别索引)
    y_pred_proba: 模型预测的概率分布
    """
    n = len(y_true)
    # 防止 log(0)
    y_pred_proba = np.clip(y_pred_proba, 1e-15, 1 - 1e-15)
    
    if y_true.ndim == 1:  # 类别索引形式
        return -np.mean(np.log(y_pred_proba[np.arange(n), y_true]))
    else:  # one-hot 形式
        return -np.mean(np.sum(y_true * np.log(y_pred_proba), axis=1))

# 示例：3 分类问题
y_true = np.array([0, 1, 2, 0])  # 真实类别
y_pred_good = np.array([
    [0.9, 0.05, 0.05],  # 预测正确，高置信度
    [0.1, 0.8, 0.1],    # 预测正确
    [0.1, 0.1, 0.8],    # 预测正确
    [0.7, 0.2, 0.1],    # 预测正确，中等置信度
])
y_pred_bad = np.array([
    [0.2, 0.7, 0.1],    # 预测错误！
    [0.8, 0.1, 0.1],    # 预测错误！
    [0.1, 0.1, 0.8],    # 预测正确
    [0.1, 0.7, 0.2],    # 预测错误！
])

ce_good = cross_entropy(y_true, y_pred_good)
ce_bad = cross_entropy(y_true, y_pred_bad)

print(f"好预测的交叉熵: {ce_good:.4f}")
print(f"坏预测的交叉熵: {ce_bad:.4f}")
print(f"\n💡 交叉熵越小 = 预测分布越接近真实分布")
print(f"   训练时最小化交叉熵 = 让模型输出逼近真实标签")
```
