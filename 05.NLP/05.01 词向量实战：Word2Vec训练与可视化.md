## 1️⃣ 词向量直觉：国王 - 男人 + 女人 ≈ 王后

```python
import numpy as np
import matplotlib.pyplot as plt
from sklearn.decomposition import PCA

# 模拟词向量（实际训练出来的是高维的，这里降到 2D 理解直觉）
words = ['king', 'queen', 'man', 'woman', 'prince', 'princess',
         'dog', 'cat', 'animal', 'apple', 'banana', 'fruit']
vectors_2d = {
    'king': [0.5, 0.8], 'queen': [0.5, 0.6], 'man': [0.3, 0.7],
    'woman': [0.3, 0.5], 'prince': [0.5, 0.9], 'princess': [0.5, 0.7],
    'dog': [-0.5, 0.8], 'cat': [-0.3, 0.8], 'animal': [-0.4, 0.6],
    'apple': [-0.5, -0.5], 'banana': [-0.3, -0.5], 'fruit': [-0.4, -0.7],
}

# 验证类比：king - man + woman ≈ queen
king = np.array(vectors_2d['king'])
man = np.array(vectors_2d['man'])
woman = np.array(vectors_2d['woman'])
queen = np.array(vectors_2d['queen'])
result = king - man + woman

print(f"king - man + woman = {result}")
print(f"queen                = {queen}")
print(f"余弦相似度: {np.dot(result, queen) / (np.linalg.norm(result) * np.linalg.norm(queen)):.3f}")

# 可视化
fig, ax = plt.subplots(figsize=(8, 8))
for word, vec in vectors_2d.items():
    ax.scatter(vec[0], vec[1], s=100)
    ax.annotate(word, (vec[0], vec[1]), xytext=(5, 5), textcoords='offset points', fontsize=12)
ax.axhline(0, color='gray', alpha=0.3); ax.axvline(0, color='gray', alpha=0.3)
ax.set_title('Word Vectors: 语义相近 → 位置相近')
ax.set_xlabel('Dimension 1'); ax.set_ylabel('Dimension 2')
ax.grid(True, alpha=0.3); plt.show()
```

## 2️⃣ 🔥 用 Gensim 训练自己的 Word2Vec

```python
# pip install gensim
from gensim.models import Word2Vec
from nltk.tokenize import sent_tokenize, word_tokenize
import nltk
nltk.download('punkt', quiet=True)

# 使用内置语料（或用你自己的文本文件）
corpus = [
    "the king ruled the kingdom with wisdom".split(),
    "the queen lived in the castle with the king".split(),
    "the prince and princess played in the garden".split(),
    "the man walked his dog in the park".split(),
    "the woman fed the cat some fish".split(),
    "apple and banana are delicious fruits".split(),
    "fruit is healthy food for humans and animals".split(),
    "the dog chased the cat up the tree".split(),
    "the cat ate the fish from the woman".split(),
]

# 训练
model = Word2Vec(sentences=corpus, vector_size=50, window=3, min_count=1, epochs=100)

# 找相似词
print("与 'king' 最相似的词:")
for word, score in model.wv.most_similar('king', topn=5):
    print(f"  {word}: {score:.3f}")

print("\n类比: king - man + woman = ?")
result = model.wv.most_similar(positive=['king', 'woman'], negative=['man'], topn=1)
print(f"  {result[0][0]} (得分: {result[0][1]:.3f})")

# 查看一个词的向量
print(f"\n'king' 的向量（前 5 维）: {model.wv['king'][:5]}")
```

> 💡 小数据集上效果有限。真实项目中用 Wikipedia 或 Common Crawl 预训练好的词向量（GloVe/FastText）。

## 3️⃣ CBOW vs Skip-gram

```
CBOW（连续词袋）:            Skip-gram:
上下文 → 目标词               目标词 → 上下文

"the ___ sat on the mat"    "cat" → 预测周围词
     ↓                           ↓
   "cat"                    "the", "sat", "on", "the"

CBOW 更快，适合大数据集。Skip-gram 对罕见词更好。
```

```python
# Gensim 中切换只需改一个参数
cbow_model = Word2Vec(corpus, sg=0)    # 0 = CBOW
skip_model = Word2Vec(corpus, sg=1)    # 1 = Skip-gram
```
