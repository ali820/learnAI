## 1️⃣ RAG 数据流全景

```
文档 → 分块 → Embedding → 向量数据库
                              ↓
用户问题 → Embedding → 检索 Top-K 文档块
                              ↓
                    [问题 + 上下文] → LLM → 回答
```

```python

import chromadb
from chromadb.utils import embedding_functions
import numpy as np

# 初始化
client = chromadb.Client()
sentence_transformer_ef = embedding_functions.SentenceTransformerEmbeddingFunction(
    model_name="all-MiniLM-L6-v2"  # 轻量 Embedding 模型，384 维
)
```

## 2️⃣ 文档分块：Chunking 策略

```python
def chunk_text(text, chunk_size=500, overlap=50):
    """滑窗分块：块与块之间有重叠，防止信息在边界断裂"""
    chunks = []
    start = 0
    while start < len(text):
        end = min(start + chunk_size, len(text))
        chunks.append(text[start:end])
        start += (chunk_size - overlap)
    return chunks

# 演示
document = """[一段很长的技术文档...]""" * 10
chunks = chunk_text(document, chunk_size=200, overlap=30)
print(f"分块数: {len(chunks)}")
for i, chunk in enumerate(chunks[:3]):
    print(f"\n块 {i} ({len(chunk)} 字): {chunk[:80]}...")

print("\n💡 Chunk Size 权衡:")
print("   太大 → 检索精度下降 → 回答不聚焦")
print("   太小 → 丢失上下文 → 回答碎片化")
print("   Overlap = 让相邻块共享边界信息")
```

## 3️⃣ 🔥 完整 RAG Pipeline

```python
# === 1. 准备知识库 ===
documents = [
    "产品A 的价格是 999 元，支持 7 天无理由退货。",
    "产品A 的保修期为 2 年，电池保修 1 年。",
    "产品B 的价格是 1499 元，不支持退货但支持换货。",
    "公司地址在北京市海淀区中关村，工作时间 9:00-18:00。",
    "投诉电话是 400-123-4567，售后邮箱 support@example.com。",
]

# 创建 collection
collection = client.create_collection(
    name="helpdesk",
    embedding_function=sentence_transformer_ef
)

# 写入向量库
collection.add(
    documents=documents,
    ids=[f"doc_{i}" for i in range(len(documents))]
)
print(f"已索引 {collection.count()} 条文档")

# === 2. 检索 ===
query = "产品A 多少钱？可以退货吗？"
results = collection.query(
    query_texts=[query],
    n_results=3
)

print(f"\n问题: {query}")
print("检索到的相关文档:")
for i, (doc, dist) in enumerate(zip(results['documents'][0], results['distances'][0])):
    print(f"  {i+1}. [相似度: {1-dist:.3f}] {doc}")

# === 3. 构建 Prompt + 生成 ===
context = "\n".join(results['documents'][0])
prompt = f"""根据以下参考信息回答问题。如果信息不足以回答，请说明。

参考信息：
{context}

问题：{query}

回答："""

# response = call_llm(prompt)
print(f"\n===== 发送给 LLM 的 Prompt =====")
print(prompt[:500])
```
