## 2️⃣ 🔥 U-Net 架构：从零实现

```
U-Net = 对称的 Encoder-Decoder + Skip Connections

Encoder（下采样）:          Decoder（上采样）:
[1,64,256,256]              [1,64,256,256]  ← 输出
    ↓ 卷积+池化                  ↑ 上采样+卷积
[64,128,128,128]    ──→    [128,128,128,128]
    ↓                           ↑
[128,256,64,64]     ──→    [256,256,64,64]
    ↓                           ↑
[256,512,32,32]    ──→     [512,512,32,32]   ← bottleneck

Skip Connection: 把 Encoder 的特征直接拼接到 Decoder 对应层
→ 保留空间细节（否则上采样会糊掉）
```

```python
import torch
import torch.nn as nn

class DoubleConv(nn.Module):
    """(Conv → BN → ReLU) × 2"""
    def __init__(self, in_ch, out_ch):
        super().__init__()
        self.conv = nn.Sequential(
            nn.Conv2d(in_ch, out_ch, 3, padding=1, bias=False),
            nn.BatchNorm2d(out_ch),
            nn.ReLU(inplace=True),
            nn.Conv2d(out_ch, out_ch, 3, padding=1, bias=False),
            nn.BatchNorm2d(out_ch),
            nn.ReLU(inplace=True),
        )
    def forward(self, x):
        return self.conv(x)

class UNet(nn.Module):
    def __init__(self, in_channels=3, num_classes=1):
        super().__init__()
        # Encoder
        self.enc1 = DoubleConv(in_channels, 64)
        self.enc2 = DoubleConv(64, 128)
        self.enc3 = DoubleConv(128, 256)
        self.enc4 = DoubleConv(256, 512)
        # Bottleneck
        self.bottleneck = DoubleConv(512, 1024)
        # Decoder
        self.up4 = nn.ConvTranspose2d(1024, 512, 2, stride=2)
        self.dec4 = DoubleConv(1024, 512)   # 1024 = 512(上采样) + 512(skip)
        self.up3 = nn.ConvTranspose2d(512, 256, 2, stride=2)
        self.dec3 = DoubleConv(512, 256)
        self.up2 = nn.ConvTranspose2d(256, 128, 2, stride=2)
        self.dec2 = DoubleConv(256, 128)
        self.up1 = nn.ConvTranspose2d(128, 64, 2, stride=2)
        self.dec1 = DoubleConv(128, 64)
        # 输出
        self.out_conv = nn.Conv2d(64, num_classes, 1)
        self.pool = nn.MaxPool2d(2)
    
    def forward(self, x):
        # Encoder
        e1 = self.enc1(x)
        e2 = self.enc2(self.pool(e1))
        e3 = self.enc3(self.pool(e2))
        e4 = self.enc4(self.pool(e3))
        # Bottleneck
        b = self.bottleneck(self.pool(e4))
        # Decoder with skip connections
        d4 = self.dec4(torch.cat([self.up4(b), e4], dim=1))
        d3 = self.dec3(torch.cat([self.up3(d4), e3], dim=1))
        d2 = self.dec2(torch.cat([self.up2(d3), e2], dim=1))
        d1 = self.dec1(torch.cat([self.up1(d2), e1], dim=1))
        # Output
        return self.out_conv(d1)

model = UNet(in_channels=3, num_classes=1)
x = torch.randn(2, 3, 256, 256)
out = model(x)
print(f"U-Net: {x.shape} → {out.shape}")
print(f"参数量: {sum(p.numel() for p in model.parameters()):,}")
```

## 3️⃣ 分割的评估指标

```python
# IoU（分割版）= 预测为类c的像素 ∩ 真实为类c的像素 / 并集
# Dice = 2 × |A ∩ B| / (|A| + |B|)  — 常用损失函数

def dice_score(pred, target, smooth=1e-6):
    """Dice Coefficient"""
    pred = pred.view(-1)
    target = target.view(-1)
    intersection = (pred * target).sum()
    return (2. * intersection + smooth) / (pred.sum() + target.sum() + smooth)

print("\n💡 Dice Loss = 1 - Dice Score")
print("   分割常用 Dice Loss + CrossEntropy 的组合损失")
```
