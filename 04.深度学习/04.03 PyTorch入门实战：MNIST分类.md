## 2️⃣ Autograd：自动求导 — 你再也用不着手算梯度了

```python
# 手算版（Phase 1 的做法）
x = torch.tensor(2.0, requires_grad=True)
y = x ** 2
y.backward()
print(f"dy/dx = {x.grad}  ← 自动算出来的！")

# 线性回归用 PyTorch 重写（对比 Phase 1 的 NumPy 版）
w = torch.randn(1, requires_grad=True)
b = torch.zeros(1, requires_grad=True)
X = torch.randn(100, 1)
y = 3 * X + 2 + 0.1 * torch.randn(100, 1)

lr = 0.1
for i in range(200):
    pred = X @ w + b
    loss = ((pred - y) ** 2).mean()
    loss.backward()
    
    with torch.no_grad():      # 不追踪梯度
        w -= lr * w.grad
        b -= lr * b.grad
        w.grad.zero_()
        b.grad.zero_()

print(f"\nPyTorch 线性回归: w={w.item():.3f}, b={b.item():.3f}")
print("💡 对比 Phase 1：不用手写梯度公式了！backward() 自动搞定一切。")
```

## 3️⃣ 🔥 nn.Module：标准模型写法

```python
class MLP(nn.Module):
    """和 04.01 的 TwoLayerNet 一样的两层网络，但写法完全不同"""
    def __init__(self, input_dim=784, hidden_dim=256, num_classes=10):
        super().__init__()
        self.fc1 = nn.Linear(input_dim, hidden_dim)
        self.fc2 = nn.Linear(hidden_dim, num_classes)
        self.dropout = nn.Dropout(0.3)
    
    def forward(self, x):
        x = x.view(x.size(0), -1)      # 把 28x28 展平成 784
        x = F.relu(self.fc1(x))
        x = self.dropout(x)
        x = self.fc2(x)                 # CrossEntropyLoss 自带 softmax
        return x

model = MLP().to(device)
print(model)
print(f"\n参数量: {sum(p.numel() for p in model.parameters()):,}")
```

## 4️⃣ DataLoader：优雅地喂数据

```python
from torchvision import datasets, transforms

# 数据预处理
transform = transforms.Compose([
    transforms.ToTensor(),                    # PIL → Tensor + 归一化到 [0,1]
    transforms.Normalize((0.1307,), (0.3081,))  # MNIST 的均值和标准差
])

# 加载 MNIST
train_dataset = datasets.MNIST('./data', train=True, download=True, transform=transform)
test_dataset = datasets.MNIST('./data', train=False, transform=transform)

train_loader = DataLoader(train_dataset, batch_size=64, shuffle=True)
test_loader = DataLoader(test_dataset, batch_size=1000, shuffle=False)

# 看一眼数据
images, labels = next(iter(train_loader))
print(f"Batch shape: {images.shape}")  # [64, 1, 28, 28]
print(f"Labels: {labels[:10]}")
```

## 5️⃣ 🔥 标准训练循环（以后每个项目都用这个模板）

```python
def train_epoch(model, loader, optimizer, criterion, device):
    model.train()
    total_loss, correct, total = 0, 0, 0
    for X, y in loader:
        X, y = X.to(device), y.to(device)
        
        optimizer.zero_grad()
        output = model(X)
        loss = criterion(output, y)
        loss.backward()
        optimizer.step()
        
        total_loss += loss.item() * X.size(0)
        correct += (output.argmax(1) == y).sum().item()
        total += X.size(0)
    return total_loss / total, correct / total

@torch.no_grad()
def evaluate(model, loader, criterion, device):
    model.eval()
    total_loss, correct, total = 0, 0, 0
    for X, y in loader:
        X, y = X.to(device), y.to(device)
        output = model(X)
        loss = criterion(output, y)
        total_loss += loss.item() * X.size(0)
        correct += (output.argmax(1) == y).sum().item()
        total += X.size(0)
    return total_loss / total, correct / total

# 训练
model = MLP().to(device)
criterion = nn.CrossEntropyLoss()
optimizer = optim.Adam(model.parameters(), lr=0.001)
scheduler = optim.lr_scheduler.StepLR(optimizer, step_size=5, gamma=0.5)

epochs = 10
for epoch in range(epochs):
    train_loss, train_acc = train_epoch(model, train_loader, optimizer, criterion, device)
    test_loss, test_acc = evaluate(model, test_loader, criterion, device)
    scheduler.step()
    print(f"Epoch {epoch+1:2d} | Train Loss: {train_loss:.4f} Acc: {train_acc:.3f} | Test Acc: {test_acc:.3f}")

print(f"\n🎉 最终测试准确率: {test_acc:.3f}")
```
