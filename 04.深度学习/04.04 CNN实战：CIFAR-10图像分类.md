## 1️⃣ 卷积的本质：手动算一遍

```python
import torch
import torch.nn as nn
import torch.nn.functional as F
import numpy as np

# 手动计算一次卷积，彻底理解它在做什么
input_tensor = torch.tensor([
    [1, 2, 0, 1],
    [0, 1, 2, 3],
    [3, 0, 1, 2],
    [2, 1, 0, 3]
], dtype=torch.float32).view(1, 1, 4, 4)  # (N, C, H, W)

# Sobel 边缘检测核
kernel = torch.tensor([
    [-1, 0, 1],
    [-2, 0, 2],
    [-1, 0, 1]
], dtype=torch.float32).view(1, 1, 3, 3)

output = F.conv2d(input_tensor, kernel, padding=0)
print("输入 4×4:")
print(input_tensor[0, 0])
print("\n边缘检测核:")
print(kernel[0, 0])
print("\n卷积输出 2×2（注意边缘被检测为高值！）:")
print(output[0, 0])
```

## 2️⃣ 🔥 标准 CNN 架构

```python
class CNN(nn.Module):
    """经典 CNN：Conv → BN → ReLU → Pool 的重复块"""
    def __init__(self, num_classes=10):
        super().__init__()
        # Block 1
        self.conv1 = nn.Sequential(
            nn.Conv2d(3, 32, 3, padding=1),      # 3×32×32 → 32×32×32
            nn.BatchNorm2d(32),
            nn.ReLU(),
            nn.Conv2d(32, 32, 3, padding=1),
            nn.BatchNorm2d(32),
            nn.ReLU(),
            nn.MaxPool2d(2),                       # → 32×16×16
            nn.Dropout2d(0.2),
        )
        # Block 2
        self.conv2 = nn.Sequential(
            nn.Conv2d(32, 64, 3, padding=1),
            nn.BatchNorm2d(64),
            nn.ReLU(),
            nn.Conv2d(64, 64, 3, padding=1),
            nn.BatchNorm2d(64),
            nn.ReLU(),
            nn.MaxPool2d(2),                       # → 64×8×8
            nn.Dropout2d(0.2),
        )
        # Block 3
        self.conv3 = nn.Sequential(
            nn.Conv2d(64, 128, 3, padding=1),
            nn.BatchNorm2d(128),
            nn.ReLU(),
            nn.MaxPool2d(2),                       # → 128×4×4
            nn.Dropout2d(0.3),
        )
        # 分类头
        self.classifier = nn.Sequential(
            nn.Flatten(),
            nn.Linear(128 * 4 * 4, 256),
            nn.ReLU(),
            nn.Dropout(0.5),
            nn.Linear(256, num_classes),
        )
    
    def forward(self, x):
        x = self.conv1(x)
        x = self.conv2(x)
        x = self.conv3(x)
        return self.classifier(x)

model = CNN()
print(f"参数量: {sum(p.numel() for p in model.parameters()):,}")
```

## 3️⃣ 数据增强：让模型更鲁棒

```python
import torchvision
from torchvision import transforms

# 训练时：随机翻转 + 裁剪 + 颜色抖动
train_transform = transforms.Compose([
    transforms.RandomHorizontalFlip(),
    transforms.RandomCrop(32, padding=4),
    transforms.ColorJitter(brightness=0.2, contrast=0.2),
    transforms.ToTensor(),
    transforms.Normalize((0.4914, 0.4822, 0.4465), (0.2470, 0.2435, 0.2616)),
])

# 测试时：只归一化
test_transform = transforms.Compose([
    transforms.ToTensor(),
    transforms.Normalize((0.4914, 0.4822, 0.4465), (0.2470, 0.2435, 0.2616)),
])

train_dataset = torchvision.datasets.CIFAR10('./data', train=True, download=True, transform=train_transform)
test_dataset = torchvision.datasets.CIFAR10('./data', train=False, transform=test_transform)
```

## 5️⃣ 🔥 迁移学习：拿 ResNet 直接微调

```python
from torchvision.models import resnet18, ResNet18_Weights

# ImageNet 预训练的 ResNet18
model_resnet = resnet18(weights=ResNet18_Weights.IMAGENET1K_V1)

# 冻结卷积层
for param in model_resnet.parameters():
    param.requires_grad = False

# 替换最后一层（ImageNet 有 1000 类 → CIFAR-10 有 10 类）
model_resnet.fc = nn.Linear(512, 10)
model_resnet = model_resnet.to(device)

# 只需训练最后一层！
print(f"可训练参数: {sum(p.numel() for p in model_resnet.parameters() if p.requires_grad):,}")
print(f"冻结的参数: {sum(p.numel() for p in model_resnet.parameters() if not p.requires_grad):,}")
print("\n💡 迁移学习 = 站在巨人肩膀上。用小数据也能达到高准确率。")
```
