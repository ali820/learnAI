## 1️⃣ 🔥 Self-Attention：核心中的核心

```python
import torch
import torch.nn as nn
import torch.nn.functional as F
import numpy as np
import math
import matplotlib.pyplot as plt
import seaborn as sns

def scaled_dot_product_attention(Q, K, V, mask=None):
    """
    Attention(Q, K, V) = softmax(QK^T / sqrt(d_k)) V
    
    Q: Query  — 我要查什么？
    K: Key    — 我能匹配什么？
    V: Value  — 匹配后拿什么信息？
    """
    d_k = Q.size(-1)
    scores = Q @ K.transpose(-2, -1) / math.sqrt(d_k)  # 缩放
    
    if mask is not None:
        scores = scores.masked_fill(mask == 0, -1e9)
    
    attention_weights = F.softmax(scores, dim=-1)
    output = attention_weights @ V
    
    return output, attention_weights

# 可视化：Self-Attention 如何让每个词"看到"所有其他词
seq_len, d_model = 6, 4
Q = K = V = torch.randn(1, seq_len, d_model)

output, weights = scaled_dot_product_attention(Q, K, V)

fig, ax = plt.subplots(figsize=(6, 5))
sns.heatmap(weights[0].detach().numpy(), annot=True, fmt='.2f', cmap='YlOrRd',
            xticklabels=['我','爱','机器','学习','。','<eos>'],
            yticklabels=['我','爱','机器','学习','。','<eos>'])
ax.set_title('Self-Attention: 每个词关注所有词')
plt.show()

print("💡 Q=K=V 来自同一个输入 → Self-Attention")
print("   Q 来自 Decoder, K/V 来自 Encoder → Cross-Attention")
print("   (Transformer Decoder 中两种都用)")
```

## 2️⃣ Multi-Head Attention：让模型从多个角度"看"

```python
class MultiHeadAttention(nn.Module):
    def __init__(self, d_model, num_heads):
        super().__init__()
        assert d_model % num_heads == 0
        
        self.d_model = d_model
        self.num_heads = num_heads
        self.d_k = d_model // num_heads
        
        self.W_Q = nn.Linear(d_model, d_model)
        self.W_K = nn.Linear(d_model, d_model)
        self.W_V = nn.Linear(d_model, d_model)
        self.W_O = nn.Linear(d_model, d_model)
    
    def split_heads(self, x):
        batch_size, seq_len, _ = x.size()
        return x.view(batch_size, seq_len, self.num_heads, self.d_k).transpose(1, 2)
        # → [batch, num_heads, seq_len, d_k]
    
    def forward(self, Q, K, V, mask=None):
        Q = self.split_heads(self.W_Q(Q))
        K = self.split_heads(self.W_K(K))
        V = self.split_heads(self.W_V(V))
        
        attn_output, attn_weights = scaled_dot_product_attention(Q, K, V, mask)
        
        # 合并多头
        attn_output = attn_output.transpose(1, 2).contiguous().view(
            Q.size(0), -1, self.d_model
        )
        
        return self.W_O(attn_output), attn_weights

# 测试
mha = MultiHeadAttention(d_model=512, num_heads=8)
x = torch.randn(2, 10, 512)  # batch=2, seq=10, dim=512
out, weights = mha(x, x, x)
print(f"Multi-Head Attention 输出: {out.shape}")  # [2, 10, 512]
print(f"每个头的权重: {weights.shape}")           # [2, 8, 10, 10]
```

## 3️⃣ Positional Encoding：注入位置信息

```python
class PositionalEncoding(nn.Module):
    def __init__(self, d_model, max_len=5000):
        super().__init__()
        pe = torch.zeros(max_len, d_model)
        position = torch.arange(0, max_len).unsqueeze(1).float()
        div_term = torch.exp(torch.arange(0, d_model, 2).float() * 
                            -(math.log(10000.0) / d_model))
        
        pe[:, 0::2] = torch.sin(position * div_term)  # 偶数维：sin
        pe[:, 1::2] = torch.cos(position * div_term)  # 奇数维：cos
        
        self.register_buffer('pe', pe.unsqueeze(0))  # [1, max_len, d_model]
    
    def forward(self, x):
        return x + self.pe[:, :x.size(1)]

# 可视化
d_model, max_len = 128, 50
pe = PositionalEncoding(d_model, max_len)
pe_matrix = pe.pe[0, :max_len, :64].numpy()  # 只看前 64 维

fig, ax = plt.subplots(figsize=(10, 6))
im = ax.imshow(pe_matrix.T, aspect='auto', cmap='RdBu')
ax.set_xlabel('Position'); ax.set_ylabel('Dimension')
ax.set_title('Positional Encoding: 每个位置有唯一"指纹"')
plt.colorbar(im); plt.show()

print("💡 为什么用 sin/cos？")
print("   1. 值在 [-1,1]，和 embedding 尺度匹配")
print("   2. sin(α+β) = sinα·cosβ + cosα·sinβ → 模型可以学到相对位置")
```

## 4️⃣ 🔥 Transformer Encoder 模块

```python
class TransformerEncoderLayer(nn.Module):
    def __init__(self, d_model, num_heads, d_ff, dropout=0.1):
        super().__init__()
        self.self_attn = MultiHeadAttention(d_model, num_heads)
        self.feed_forward = nn.Sequential(
            nn.Linear(d_model, d_ff),
            nn.ReLU(),
            nn.Linear(d_ff, d_model),
        )
        self.norm1 = nn.LayerNorm(d_model)
        self.norm2 = nn.LayerNorm(d_model)
        self.dropout = nn.Dropout(dropout)
    
    def forward(self, x, mask=None):
        # Self-Attention + Residual + LayerNorm
        attn_out, _ = self.self_attn(x, x, x, mask)
        x = self.norm1(x + self.dropout(attn_out))
        
        # Feed-Forward + Residual + LayerNorm
        ff_out = self.feed_forward(x)
        x = self.norm2(x + self.dropout(ff_out))
        
        return x

# 搭建一个微型 Transformer
class MiniTransformer(nn.Module):
    def __init__(self, vocab_size, d_model=256, num_heads=4, num_layers=3, d_ff=512):
        super().__init__()
        self.embedding = nn.Embedding(vocab_size, d_model)
        self.pos_encoding = PositionalEncoding(d_model)
        self.encoder_layers = nn.ModuleList([
            TransformerEncoderLayer(d_model, num_heads, d_ff) for _ in range(num_layers)
        ])
        self.fc_out = nn.Linear(d_model, vocab_size)
    
    def forward(self, x):
        x = self.embedding(x)
        x = self.pos_encoding(x)
        for layer in self.encoder_layers:
            x = layer(x)
        return self.fc_out(x)

model = MiniTransformer(vocab_size=1000)
print(f"Transformer 参数量: {sum(p.numel() for p in model.parameters()):,}")
```

## 5️⃣ Transformer vs RNN 对比

| | RNN/LSTM | Transformer |
|---|---|---|
| 并行性 | ❌ 必须逐个时间步 | ✅ 所有位置同时计算 |
| 长距离依赖 | ⚠️ LSTM 缓解但不根治 | ✅ 任意位置 O(1) 连接 |
| 训练速度 | 慢 | 快（特别是长序列） |
| 可解释性 | ✅ Attention weights 可视化 | ✅ Attention weights 可视化 |
| 复杂度 | O(n) | O(n²) — 注意力矩阵 n×n |

> 💡 Transformer 的 O(n²) 是限制长上下文的主要瓶颈。Flash Attention、Mamba 等后续工作都在解决这个问题。
