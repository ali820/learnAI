# 决策树与随机森林实战

> 决策树是"白话版"模型——每个判断都能用人话解释。随机森林把它升级成了稳定版。

---

## 🎯 练习目标

1. 理解决策树的分裂原理（信息增益/基尼系数）
2. 掌握随机森林的 Bagging 思想
3. 学会可视化决策树
4. 对比单棵树和森林的性能

---

## 1️⃣ 决策树：可视化看"怎么分裂"

```python
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.datasets import load_iris
from sklearn.tree import DecisionTreeClassifier, plot_tree

# 只用 2 个特征以便可视化
iris = load_iris()
X = iris.data[:, :2]  # sepal length, sepal width
y = iris.target
feature_names = iris.feature_names[:2]
class_names = iris.target_names

# 训练一棵小树（限制深度方便画图）
tree = DecisionTreeClassifier(max_depth=3, random_state=42)
tree.fit(X, y)

# 🌲 画出决策树
plt.figure(figsize=(12, 7))
plot_tree(tree, feature_names=feature_names, class_names=class_names,
          filled=True, rounded=True, fontsize=10)
plt.title('Decision Tree (max_depth=3)')
plt.show()

print(f"训练准确率: {tree.score(X, y):.3f}")
```

## 2️⃣ 决策边界对比

```python
from sklearn.ensemble import RandomForestClassifier
from matplotlib.colors import ListedColormap

def plot_decision_boundary(ax, model, X, y, title):
    x_min, x_max = X[:, 0].min() - 0.5, X[:, 0].max() + 0.5
    y_min, y_max = X[:, 1].min() - 0.5, X[:, 1].max() + 0.5
    xx, yy = np.meshgrid(np.linspace(x_min, x_max, 300),
                         np.linspace(y_min, y_max, 300))
    Z = model.predict(np.c_[xx.ravel(), yy.ravel()])
    Z = Z.reshape(xx.shape)
    
    cmap_background = ListedColormap(['#FFAAAA', '#AAFFAA', '#AAAAFF'])
    ax.contourf(xx, yy, Z, alpha=0.3, cmap=cmap_background)
    ax.scatter(X[:, 0], X[:, 1], c=y, edgecolor='k', s=30, cmap=plt.cm.Set1)
    ax.set_xlabel(feature_names[0])
    ax.set_ylabel(feature_names[1])
    ax.set_title(title)

fig, axes = plt.subplots(1, 3, figsize=(18, 5))

# 浅树
tree1 = DecisionTreeClassifier(max_depth=2, random_state=42).fit(X, y)
plot_decision_boundary(axes[0], tree1, X, y, 'Decision Tree (depth=2)')

# 深树（过拟合）
tree2 = DecisionTreeClassifier(max_depth=None, random_state=42).fit(X, y)
plot_decision_boundary(axes[1], tree2, X, y, 'Decision Tree (No Limit → Overfit)')

# 随机森林
rf = RandomForestClassifier(n_estimators=100, max_depth=3, random_state=42).fit(X, y)
plot_decision_boundary(axes[2], rf, X, y, 'Random Forest (100 trees, depth=3)')

plt.tight_layout()
plt.show()
```

> 💡 单棵树太深 → 边界弯弯曲曲（过拟合）。随机森林 100 棵树投票 → 平滑稳定。

## 3️⃣ 随机森林：集成学习的力量

```python
from sklearn.ensemble import RandomForestClassifier
from sklearn.model_selection import cross_val_score
from sklearn.metrics import accuracy_score

# 加载完整 Iris（4 个特征）
X_full, y_full = load_iris(return_X_y=True)
X_train, X_test, y_train, y_test = train_test_split(
    X_full, y_full, test_size=0.3, random_state=42
)

# 对比：单棵树 vs 随机森林
from sklearn.model_selection import train_test_split

models = {
    'Decision Tree': DecisionTreeClassifier(random_state=42),
    'Random Forest (10)': RandomForestClassifier(n_estimators=10, random_state=42),
    'Random Forest (100)': RandomForestClassifier(n_estimators=100, random_state=42),
}

print("=== 模型对比（5-fold CV）===")
for name, model in models.items():
    cv_scores = cross_val_score(model, X_full, y_full, cv=5)
    model.fit(X_train, y_train)
    test_acc = accuracy_score(y_test, model.predict(X_test))
    print(f"{name:20s}  CV={cv_scores.mean():.3f}±{cv_scores.std():.3f}  Test={test_acc:.3f}")
```

## 4️⃣ 特征重要性分析

```python
rf = RandomForestClassifier(n_estimators=100, random_state=42)
rf.fit(X_train, y_train)

importance_df = pd.DataFrame({
    'Feature': iris.feature_names,
    'Importance': rf.feature_importances_
}).sort_values('Importance', ascending=False)

plt.figure(figsize=(8, 5))
plt.barh(importance_df['Feature'], importance_df['Importance'], color='green', alpha=0.7)
plt.xlabel('Feature Importance')
plt.title('Random Forest Feature Importance')
plt.gca().invert_yaxis()
plt.grid(True, alpha=0.3)
plt.show()

print(importance_df)
```

> 💡 特征重要性 = 这个特征在所有树的所有分裂中"帮了多少忙"。不是因果分析，但能帮你快速筛选有用特征。
