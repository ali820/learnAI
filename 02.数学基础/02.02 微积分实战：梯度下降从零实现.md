# 微积分实战：梯度下降从零实现

> **核心直觉**：梯度下降 = 蒙着眼下山，每次往脚下最陡的方向走一步。
> 理解了它，你就理解了 90% 的深度学习训练过程。

---

## 🎯 练习目标

1. 用 NumPy 从零实现梯度下降
2. 可视化梯度下降的过程（直观理解"最陡方向"）
3. 手动验证链式法则（反向传播的前置知识）
4. 理解学习率、震荡、收敛的关系

---

## 1️⃣ 导数 = 变化率 = 方向

```python
import numpy as np
import matplotlib.pyplot as plt

def f(x):
    """最简单的二次函数"""
    return x**2

def df(x):
    """f(x) = x² 的导数：f'(x) = 2x"""
    return 2 * x

# 可视化导数的几何意义
x = np.linspace(-3, 3, 100)
fig, ax = plt.subplots(figsize=(8, 5))
ax.plot(x, f(x), 'b-', linewidth=2, label='f(x) = x²')

# 在 x₀ 处画切线
x0 = 2
slope = df(x0)          # 斜率 = 4
tangent = f(x0) + slope * (x - x0)
ax.plot(x, tangent, 'r--', linewidth=2, label=f'切线 (斜率={slope})')
ax.plot(x0, f(x0), 'ro', markersize=8)

ax.set_xlabel('x')
ax.set_ylabel('f(x)')
ax.legend()
ax.grid(True, alpha=0.3)
ax.set_title('导数 = 切线的斜率')
plt.show()

print(f"f({x0}) = {f(x0)}")
print(f"导数 f'({x0}) = {df(x0)}  ← 正值意味着往右走 f 增大")
```

> 💡 导数的正负告诉你"往哪走函数会增大"。**梯度的反方向**就是函数值下降最快的方向。

## 2️⃣ 梯度下降 — 一维版

从 `x = -2.5` 出发，一步步走向最小值 `x = 0`：

```python
def gradient_descent_1d(start, lr=0.1, steps=20):
    """一维梯度下降"""
    x = start
    history = [x]
    
    for i in range(steps):
        grad = df(x)          # 计算梯度
        x = x - lr * grad     # 往梯度的反方向走
        history.append(x)
    
    return np.array(history)

# 运行
history = gradient_descent_1d(start=-2.5, lr=0.1, steps=20)

# 可视化下降过程
x = np.linspace(-3, 3, 100)
fig, ax = plt.subplots(figsize=(10, 5))
ax.plot(x, f(x), 'b-', linewidth=2, label='f(x) = x²')
ax.plot(history, f(history), 'ro-', markersize=4, label='梯度下降路径')
ax.plot(history[0], f(history[0]), 'go', markersize=10, label='起点')
ax.plot(history[-1], f(history[-1]), 'r*', markersize=15, label=f'终点 x≈{history[-1]:.3f}')
ax.set_xlabel('x')
ax.set_ylabel('f(x)')
ax.legend()
ax.grid(True, alpha=0.3)
ax.set_title('梯度下降：一步步走向最低点')
plt.show()

print(f"起点: x={history[0]:.3f}, f={f(history[0]):.3f}")
print(f"终点: x={history[-1]:.6f}, f={f(history[-1]):.6f}")
```

> 💡 **这就是训练神经网络的核心循环**：算梯度 → 往反方向走一步 → 重复。

## 3️⃣ 学习率的影响 — 必看！

```python
fig, axes = plt.subplots(1, 3, figsize=(15, 4))
learning_rates = [0.01, 0.1, 1.01]  # 太小、合适、太大
titles = ['lr=0.01（太慢）', 'lr=0.1（合适）', 'lr=1.01（震荡发散！）']

for ax, lr, title in zip(axes, learning_rates, titles):
    history = gradient_descent_1d(start=-2.5, lr=lr, steps=15)
    ax.plot(history, f(history), 'ro-', markersize=4)
    ax.plot(history[0], f(history[0]), 'go', markersize=8)
    ax.set_xlabel('x')
    ax.set_ylabel('f(x)')
    ax.set_title(title)
    ax.grid(True, alpha=0.3)

plt.tight_layout()
plt.show()
```

> 💡 **学习率是深度学习中最重要也最难调的超参数。** 太小 → 训练慢；太大 → 震荡发散。
> Phase 3 学 Adam 优化器时会知道：自适应学习率自动避免了这个问题。

## 4️⃣ 多元函数与偏导数 → 真正的梯度下降

```python
def f_2d(x, y):
    """二元函数：f(x,y) = x² + y² （一个碗的形状）"""
    return x**2 + y**2

def grad_2d(x, y):
    """梯度 = [∂f/∂x, ∂f/∂y] = [2x, 2y]"""
    return np.array([2*x, 2*y])

def gradient_descent_2d(start, lr=0.1, steps=30):
    """二维梯度下降"""
    pos = np.array(start, dtype=float)
    history = [pos.copy()]
    
    for _ in range(steps):
        grad = grad_2d(*pos)
        pos -= lr * grad
        history.append(pos.copy())
    
    return np.array(history)

history_2d = gradient_descent_2d(start=[-2.5, 2.0], lr=0.2, steps=15)

# 画等高线和下降路径
x = np.linspace(-3, 3, 50)
y = np.linspace(-3, 3, 50)
X, Y = np.meshgrid(x, y)
Z = f_2d(X, Y)

fig, ax = plt.subplots(figsize=(8, 7))
ax.contour(X, Y, Z, levels=20, cmap='Blues', alpha=0.6)
ax.plot(history_2d[:, 0], history_2d[:, 1], 'ro-', markersize=4, linewidth=2)
ax.plot(history_2d[0, 0], history_2d[0, 1], 'go', markersize=10, label='起点')
ax.plot(history_2d[-1, 0], history_2d[-1, 1], 'r*', markersize=15, label='最低点')
ax.set_xlabel('x')
ax.set_ylabel('y')
ax.legend()
ax.set_title('二维梯度下降：垂直于等高线方向下降')
ax.set_aspect('equal')
plt.show()
```

> 💡 等高线上，下降路径始终**垂直于等高线**——这就是梯度方向的几何意义。

## 5️⃣ 🔥 核心实战：线性回归的梯度下降求解

```python
# 生成模拟数据：y = 3x + 2 + 噪声
np.random.seed(42)
X = np.random.randn(100, 1)
true_w, true_b = 3.0, 2.0
y = true_w * X + true_b + np.random.randn(100, 1) * 0.5

# 梯度下降求解线性回归
def mse_loss(X, y, w, b):
    """均方误差"""
    pred = X @ w + b
    return np.mean((pred - y) ** 2)

def mse_grad(X, y, w, b):
    """MSE 对 w 和 b 的梯度"""
    N = len(X)
    pred = X @ w + b
    error = pred - y
    dw = (2 / N) * (X.T @ error)
    db = (2 / N) * np.sum(error)
    return dw, db

# 训练
w, b = np.random.randn(1, 1), 0.0  # 随机初始化
lr = 0.1
history_loss = []

for i in range(200):
    loss = mse_loss(X, y, w, b)
    dw, db = mse_grad(X, y, w, b)
    w -= lr * dw
    b -= lr * db
    history_loss.append(loss)

print(f"训练结果: w = {w[0,0]:.3f} (真值 {true_w}), b = {b:.3f} (真值 {true_b})")

# 可视化损失下降
fig, axes = plt.subplots(1, 2, figsize=(12, 4))
axes[0].plot(history_loss)
axes[0].set_xlabel('迭代次数')
axes[0].set_ylabel('MSE 损失')
axes[0].set_title('损失函数收敛过程')
axes[0].grid(True, alpha=0.3)

axes[1].scatter(X, y, alpha=0.5, s=10, label='数据')
x_line = np.linspace(-3, 3, 100).reshape(-1, 1)
axes[1].plot(x_line, x_line * w + b, 'r-', linewidth=2, label=f'y={w[0,0]:.2f}x+{b:.2f}')
axes[1].set_xlabel('X')
axes[1].set_ylabel('y')
axes[1].legend()
axes[1].set_title('拟合结果')
axes[1].grid(True, alpha=0.3)
plt.tight_layout()
plt.show()
```

> 💡 这就是 **Phase 2 机器学习的核心训练逻辑**。``PyTorch``/``TensorFlow`` 只是帮你自动算梯度（Autograd），本质还是这个循环！

## 6️⃣ 链式法则手工验证（预习反向传播）

```python
# 复合函数：h(x) = g(f(x))
# 其中 f(x) = x²，g(z) = sin(z)
# h'(x) = g'(f(x)) · f'(x) = cos(x²) · 2x

def f(x): return x**2
def df(x): return 2*x
def g(z): return np.sin(z)
def dg(z): return np.cos(z)

x = 1.5

# 手工链式法则
manual = dg(f(x)) * df(x)  # cos(2.25) * 3.0

# NumPy 数值验证
h = 1e-6
numerical = (g(f(x + h)) - g(f(x - h))) / (2 * h)

print(f"链式法则: {manual:.6f}")
print(f"数值验证: {numerical:.6f}")
print(f"误差: {abs(manual - numerical):.2e}")
```

> 💡 **链式法则 = 反向传播的数学基础。** 神经网络有多层，梯度一层层往回荡，每层都是链式法则的一次应用。Phase 3 学 PyTorch 的 ``.backward()`` 时，你会在计算图上看到一模一样的事。

# 试试不同学习率，观察损失曲线的不同
lrs = [0.001, 0.01, 0.05, 0.1, 0.5]
