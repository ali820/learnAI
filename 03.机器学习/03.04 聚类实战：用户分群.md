## 1️⃣ K-Means 从零实现

```python
import numpy as np
import matplotlib.pyplot as plt

class KMeansFromScratch:
    def __init__(self, k=3, max_iter=100):
        self.k = k
        self.max_iter = max_iter
    
    def fit(self, X):
        n, d = X.shape
        
        # 随机初始化中心点
        idx = np.random.choice(n, self.k, replace=False)
        self.centers = X[idx].copy()
        
        for _ in range(self.max_iter):
            # Step 1: 分配每个点到最近的中心
            distances = np.linalg.norm(X[:, None] - self.centers[None, :], axis=2)
            labels = np.argmin(distances, axis=1)
            
            # Step 2: 更新中心点
            new_centers = np.array([X[labels == j].mean(axis=0) for j in range(self.k)])
            
            if np.allclose(self.centers, new_centers):
                break
            self.centers = new_centers
        
        self.labels_ = np.argmin(
            np.linalg.norm(X[:, None] - self.centers[None, :], axis=2), axis=1)
        return self

# 生成测试数据
from sklearn.datasets import make_blobs
X, y_true = make_blobs(n_samples=300, centers=4, cluster_std=0.8, random_state=42)

kmeans = KMeansFromScratch(k=4)
kmeans.fit(X)

# 可视化
fig, axes = plt.subplots(1, 2, figsize=(14, 5))
axes[0].scatter(X[:, 0], X[:, 1], c=y_true, cmap='Set1', s=10)
axes[0].set_title('Ground Truth')
axes[1].scatter(X[:, 0], X[:, 1], c=kmeans.labels_, cmap='Set1', s=10)
axes[1].scatter(kmeans.centers[:, 0], kmeans.centers[:, 1],
                c='black', marker='x', s=200, linewidths=3, label='Centers')
axes[1].set_title('K-Means (k=4)')
axes[1].legend()
plt.tight_layout()
plt.show()
```

## 2️⃣ Sklearn K-Means + 肘部法则

```python
from sklearn.cluster import KMeans

# 肘部法则：找到"拐点"
inertias = []
K_range = range(1, 11)
for k in K_range:
    km = KMeans(n_clusters=k, n_init=10, random_state=42)
    km.fit(X)
    inertias.append(km.inertia_)

plt.figure(figsize=(8, 5))
plt.plot(K_range, inertias, 'bo-')
plt.xlabel('Number of Clusters (k)')
plt.ylabel('Inertia (簇内平方和)')
plt.title('肘部法则：找拐点选 K')
plt.axvline(4, color='r', linestyle='--', alpha=0.5, label='最佳 k≈4')
plt.legend()
plt.grid(True, alpha=0.3)
plt.show()

print("💡 肘部在 k=4 处拐弯 → 最佳 k=4")
```

## 3️⃣ K-Means vs DBSCAN

```python
from sklearn.cluster import DBSCAN
from sklearn.datasets import make_moons

# 月亮形数据（K-Means 搞不定的形状）
X_moons, _ = make_moons(n_samples=200, noise=0.08, random_state=42)

fig, axes = plt.subplots(1, 2, figsize=(14, 5))

# K-Means
km = KMeans(n_clusters=2, n_init=10, random_state=42).fit(X_moons)
axes[0].scatter(X_moons[:, 0], X_moons[:, 1], c=km.labels_, cmap='Set1', s=10)
axes[0].set_title('K-Means (k=2) — 不行！')

# DBSCAN
db = DBSCAN(eps=0.3, min_samples=5).fit(X_moons)
axes[1].scatter(X_moons[:, 0], X_moons[:, 1], c=db.labels_, cmap='Set1', s=10)
axes[1].set_title('DBSCAN — 完美分开了！')

plt.tight_layout()
plt.show()

print(f"K-Means 聚类数: {len(set(km.labels_))}, 噪声点: 0")
print(f"DBSCAN 聚类数: {len(set(db.labels_)) - (1 if -1 in db.labels_ else 0)}, 噪声点: {sum(db.labels_ == -1)}")
```

> 💡 K-Means 假设簇是球形的。DBSCAN 基于密度，能处理任意形状的簇，还能自动识别噪声。

## 4️⃣ PCA + t-SNE 降维可视化

```python
from sklearn.decomposition import PCA
from sklearn.manifold import TSNE
from sklearn.datasets import load_digits

# 高维数据降维到 2D 可视化
digits = load_digits()
X_digits, y_digits = digits.data, digits.target

# PCA
X_pca = PCA(n_components=2).fit_transform(X_digits)

# t-SNE（慢但效果好）
X_tsne = TSNE(n_components=2, random_state=42).fit_transform(X_digits)

fig, axes = plt.subplots(1, 2, figsize=(14, 6))
for ax, X_2d, title in [(axes[0], X_pca, 'PCA'), (axes[1], X_tsne, 't-SNE')]:
    scatter = ax.scatter(X_2d[:, 0], X_2d[:, 1], c=y_digits, cmap='tab10', s=5, alpha=0.6)
    ax.set_title(title)
plt.colorbar(scatter, ax=axes[1], label='Digit')
plt.tight_layout()
plt.show()

print("💡 PCA 保留全局结构，t-SNE 保留局部结构（同类聚得更紧）。")
```
